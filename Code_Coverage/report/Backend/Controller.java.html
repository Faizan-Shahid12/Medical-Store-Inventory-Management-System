        public String getSupName(int id) throws SQLException
        {
            ArrayList<Supplier> SupList = DB.GetAllSupplier();
            
            for(Supplier Sup: SupList)
            {
                if(Sup.getSupId() == id)
                {
                    return Sup.getName();
                }
            }
            
            return null;
        }
